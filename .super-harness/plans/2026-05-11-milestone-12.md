```typescript
import fs from 'fs'
import path from 'path'
import crypto from 'crypto'
import lockfile from 'proper-lockfile'
import type { SessionIndexEntry, Message } from '../types.js'

export class SessionManager {
  private sessionsDir: string
  private indexPath: string

  constructor(dataDir: string) {
    this.sessionsDir = path.join(dataDir, 'sessions')
    this.indexPath = path.join(this.sessionsDir, 'index.json')
    this.ensureDirectories()
  }

  private ensureDirectories(): void {
    fs.mkdirSync(this.sessionsDir, { recursive: true })
  }

  private async withIndexLock<T>(fn: () => Promise<T>): Promise<T> {
    const lockPath = path.join(this.sessionsDir, '.index.lock')
    fs.writeFileSync(lockPath, '')
    const release = await lockfile.lock(lockPath, { retries: 3, realpath: false })
    try {
      return await fn()
    } finally {
      await release()
    }
  }

  private readIndex(): SessionIndexEntry[] {
    try {
      if (fs.existsSync(this.indexPath)) {
        const content = fs.readFileSync(this.indexPath, 'utf-8')
        return JSON.parse(content)
      }
    } catch (e) {
      console.warn('Failed to read session index, rebuilding:', e)
    }
    return []
  }

  private writeIndex(entries: SessionIndexEntry[]): void {
    const tmpPath = path.join(this.sessionsDir, `.tmp-${crypto.randomUUID()}`)
    const fd = fs.openSync(tmpPath, 'w')
    fs.writeSync(fd, JSON.stringify(entries, null, 2))
    fs.fsyncSync(fd)
    fs.closeSync(fd)
    fs.renameSync(tmpPath, this.indexPath)
  }

  createSession(): string {
    const id = crypto.randomUUID()
    const now = new Date().toISOString()
    const sessionPath = path.join(this.sessionsDir, `${id}.json`)

    // 创建空 session 文件
    fs.writeFileSync(sessionPath, JSON.stringify({ id, messages: [], created: now, updated: now }))

    // 更新索引
    const entries = this.readIndex()
    entries.push({
      id,
      created: now,
      updated: now,
      message_count: 0,
      preview: '',
    })
    this.writeIndex(entries)

    return id
  }

  async saveSession(sessionId: string, messages: Message[], baseCount: number): Promise<void> {
    const sessionPath = path.join(this.sessionsDir, `${sessionId}.json`)

    await this.withIndexLock(async () => {
      // 读取现有 session
      let existing: { messages: Message[] } = { messages: [] }
      try {
        const content = fs.readFileSync(sessionPath, 'utf-8')
        existing = JSON.parse(content)
      } catch {
        // 文件不存在，使用空 session
      }

      // 合并消息（保留 baseCount 之前的消息）
      const mergedMessages = [...existing.messages.slice(0, baseCount), ...messages]

      // 原子写入
      const now = new Date().toISOString()
      const tmpPath = path.join(this.sessionsDir, `.tmp-${crypto.randomUUID()}`)
      const fd = fs.openSync(tmpPath, 'w')
      fs.writeSync(fd, JSON.stringify({
        id: sessionId,
        messages: mergedMessages,
        created: existing.messages.length > 0 ? existing.messages[0].timestamp : now,
        updated: now,
      }))
      fs.fsyncSync(fd)
      fs.closeSync(fd)
      fs.renameSync(tmpPath, sessionPath)

      // 更新索引
      const entries = this.readIndex()
      const entry = entries.find(e => e.id === sessionId)
      if (entry) {
        entry.updated = now
        entry.message_count = mergedMessages.length
        entry.preview = mergedMessages.length > 0 ? mergedMessages[mergedMessages.length - 1].content.slice(0, 100) : ''
      }
      this.writeIndex(entries)
    })
  }

  loadSession(sessionId: string): Message[] {
    const sessionPath = path.join(this.sessionsDir, `${sessionId}.json`)
    try {
      const content = fs.readFileSync(sessionPath, 'utf-8')
      const session = JSON.parse(content)
      return session.messages || []
    } catch {
      return []
    }
  }

  listSessions(): SessionIndexEntry[] {
    return this.readIndex()
  }

  async resumeSession(sessionId: string): Promise<Message[]> {
    return this.loadSession(sessionId)
  }
}
```

import type { Message } from '../types.js'
import type { SessionManager } from './session.js'

export class MemoryStorage {
  private messages: Message[] = []
  private sessionId: string | null = null
  private baseCount: number = 0

  constructor(private sessionManager: SessionManager) {}

  createSession(): string {
    this.sessionId = this.sessionManager.createSession()
    this.messages = []
    this.baseCount = 0
    return this.sessionId
  }

  addMessage(role: string, content: string): void {
    const message: Message = {
      role: role as Message['role'],
      content,
      timestamp: new Date().toISOString(),
    }
    this.messages.push(message)
  }

  async save(): Promise<void> {
    if (!this.sessionId) return
    await this.sessionManager.saveSession(this.sessionId, this.messages, this.baseCount)
    this.baseCount = this.messages.length
  }

  getMessages(): Message[] {
    return [...this.messages]
  }

  getSessionId(): string | null {
    return this.sessionId
  }

  clear(): void {
    this.messages = []
    this.sessionId = null
    this.baseCount = 0
  }
}

import type { Message } from '../types.js'
import type { SoulManager } from './soul.js'
import type { UserMemoryManager } from './user.js'

const ONBOARDING_TEMPLATE = `你是 Brix，一个个人 AI 助手。你正在与用户进行初次对话，了解他们的需求和偏好。

请通过以下步骤进行 onboarding：
1. 问候用户，介绍自己
2. 询问用户的名字或称呼
3. 询问用户的主要工作或兴趣领域
4. 询问用户希望你帮助完成什么类型的任务
5. 询问用户对 AI 助手的期望和偏好

记住：
- 保持友好和专业
- 不要一次问太多问题
- 根据用户的回答调整后续问题
- 完成后，将收集到的信息保存到 user.md`

const MEMORY_MGMT_TEMPLATE = `你是一个记忆管理助手。你的任务是帮助用户管理他们的记忆和偏好。

当前用户记忆：
{user_memory}

请根据对话内容，更新用户记忆。注意：
- 保留重要的个人信息和偏好
- 删除过时或不准确的信息
- 保持记忆简洁和有用
- 使用 Markdown 格式`

export class MemoryStrategy {
  private encoder: ReturnType<typeof encodingForModel> | null = null
  private maxTokens: number

  constructor(
    private soulManager: SoulManager,
    private userManager: UserMemoryManager,
    maxTokens: number = 8000
  ) {
    this.maxTokens = maxTokens
    try {
      this.encoder = encodingForModel('gpt-4')
    } catch {
      this.encoder = null
    }
  }

  private countTokens(text: string): number {
    if (!text) return 0
    if (this.encoder) return this.encoder.encode(text).length
    return Math.max(1, Math.floor(text.length / 4))
  }

  buildSystemPrompt(sessionContext?: string, dynamicContext?: string): string {
    const parts: string[] = []

    // Soul（无 data-guard）
    const soul = this.soulManager.load()
    if (soul) parts.push(soul)

    // User memory（with data-guard）
    const userMemory = this.userManager.load()
    if (userMemory) {
      parts.push(`## 用户记忆\n\n${userMemory}`)
    }

    // Session context
    if (sessionContext) parts.push(sessionContext)

    // Dynamic context
    if (dynamicContext) parts.push(dynamicContext)

    return parts.join('\n\n')
  }

  getContextMessages(history: Message[], systemPrompt: string): Message[] {
    const systemTokens = this.countTokens(systemPrompt)
    const availableTokens = this.maxTokens - systemTokens

    const result: Message[] = []
    let usedTokens = 0

    // 从后往前遍历 history
    for (let i = history.length - 1; i >= 0; i--) {
      const msg = history[i]
      const tokens = this.countTokens(msg.content)

      // 保留 system messages
      if (msg.role === 'system') {
        result.unshift(msg)
        continue
      }

      // 检查是否超出 token 限制
      if (usedTokens + tokens > availableTokens) break

      result.unshift(msg)
      usedTokens += tokens
    }

    return result
  }

  getOnboardingTemplate(): string {
    return ONBOARDING_TEMPLATE
  }

  getMemoryManagementTemplate(): string {
    return MEMORY_MGMT_TEMPLATE
  }
}

export class BrixMemoryProvider implements MemoryProvider {
  private sessionManager: SessionManager
  private soulManager: SoulManager
  private userManager: UserMemoryManager
  private storage: MemoryStorage
  private strategy: MemoryStrategy

  constructor(dataDir: string, maxTokens: number = 8000) {
    this.sessionManager = new SessionManager(dataDir)
    this.soulManager = new SoulManager(dataDir)
    this.userManager = new UserMemoryManager(dataDir)

    this.storage.clear()
  }

    // 懒创建 session
    if (!this.storage.getSessionId()) {
      this.storage.createSession()
    }
    this.storage.addMessage(role, content)
  }

  async saveSession(): Promise<void> {
    await this.storage.save()
  }

  }

  buildSystemPrompt(sessionContext?: string, dynamicContext?: string): string {
    return this.strategy.buildSystemPrompt(sessionContext, dynamicContext)

import crypto from 'crypto'

export class FlowLog {
  private traceId: string
  private steps: Array<{ module: string; data: Record<string, unknown>; ts: number }> = []
  private startTime: number
  private model: string = ''
  private error: string = ''

  constructor(private input: string) {
    this.traceId = crypto.randomUUID().replace(/-/g, '').slice(0, 8)
    this.startTime = performance.now()
  }

  step(module: string, data: Record<string, unknown> = {}): void {
    this.steps.push({ module, data, ts: performance.now() })
  }

  setModel(model: string): void {
    this.model = model
  }

  setError(error: string): void {
    this.error = error
  }

  finish(): Record<string, unknown> {
    const endTime = performance.now()
    return {
      trace_id: this.traceId,
      input: this.input,
      model: this.model,
      steps: this.steps.map(s => ({
        module: s.module,
        data: s.data,
        ts: s.ts - this.startTime,
      })),
      total_ms: endTime - this.startTime,
      error: this.error,
      timestamp: new Date().toISOString(),
    }
  }

  getTraceId(): string {
    return this.traceId
  }
}

import fs from 'fs'
import path from 'path'

export function writeJsonl(filePath: string, data: Record<string, unknown>): void {
  const line = JSON.stringify(data) + '\n'
  fs.appendFileSync(filePath, line)
}

export function flushLog(filePath: string, data: Record<string, unknown>): void {
  writeJsonl(filePath, data)
}

export function readAll(filePath: string): Record<string, unknown>[] {
  try {
    const content = fs.readFileSync(filePath, 'utf-8')
    return content
      .split('\n')
      .filter(line => line.trim())
      .map(line => JSON.parse(line))
  } catch {
    return []
  }
}

export function readEntry(filePath: string, traceId: string): Record<string, unknown> | null {
  const entries = readAll(filePath)
  return entries.find(e => e.trace_id === traceId) || null
}

export function formatDetail(entry: Record<string, unknown>): string {
  const steps = entry.steps as Array<{ module: string; data: Record<string, unknown>; ts: number }>
  const lines: string[] = []

  lines.push(`Trace: ${entry.trace_id}`)
  lines.push(`Input: ${entry.input}`)
  lines.push(`Model: ${entry.model}`)
  lines.push(`Total: ${(entry.total_ms as number).toFixed(1)}ms`)
  lines.push('')

  for (let i = 0; i < steps.length; i++) {
    const step = steps[i]
    const prevTs = i > 0 ? steps[i - 1].ts : 0
    const delta = step.ts - prevTs
    lines.push(`  ${step.module} (+${delta.toFixed(1)}ms)`)
    if (Object.keys(step.data).length > 0) {
      lines.push(`    ${JSON.stringify(step.data)}`)
    }
  }

  if (entry.error) {
    lines.push('')
    lines.push(`Error: ${entry.error}`)
  }

  return lines.join('\n')
}

import type { FlowLog } from '../log/flow.js'
import type { HookEvent } from '../types.js'

export class HookRegistry {
  private log: FlowLog | null = null
  private hooks: Map<string, Array<(event: HookEvent) => void | Promise<void>>> = new Map()

  bindLog(log: FlowLog): void {
    this.log = log
  }

  register(event: string, hook: (event: HookEvent) => void | Promise<void>): void {
    if (!this.hooks.has(event)) {
      this.hooks.set(event, [])
    }
    this.hooks.get(event)!.push(hook)
  }

  async fire(event: string, data: Record<string, unknown> = {}): Promise<void> {
    // 记录到 FlowLog
    if (this.log) {
      this.log.step(`hook:${event}`, data)
    }

    // 调用所有注册的 hook
    const hooks = this.hooks.get(event) || []
    for (const hook of hooks) {
      try {
        await hook({ name: event, data })
      } catch (e) {
        console.warn(`Hook ${event} failed:`, e)
      }
    }
  }
}
