```typescript
import type { LLMClient } from '../infra/llm-client.js'
import type { HookRegistry } from '../hooks/registry.js'

export type Intent = 'chat' | 'task' | 'tool_use'

const KEYWORDS_TASK = ['create', 'build', 'make', 'implement', 'fix', 'refactor', 'update', 'change', 'add', 'remove', 'delete']
const KEYWORDS_TOOL = ['calculate', 'weather', 'file', 'read', 'write', 'edit']

export async function classifyIntent(
  input: string,
  llmClient: LLMClient,
  model: string,
  hooks?: HookRegistry
): Promise<Intent> {
  // 触发 hook
  if (hooks) {
    await hooks.fire('intent', { input })
  }

  try {
    // 尝试 LLM 分类
    const response = await llmClient.chat(
      [{ role: 'user', content: `Classify the following user input as "chat", "task", or "tool_use": "${input}"` }],
      model
    )
    const intent = response.content.trim().toLowerCase() as Intent
    if (['chat', 'task', 'tool_use'].includes(intent)) {
      return intent
    }
  } catch {
    // LLM 失败，回退到关键词启发式
  }

  // 关键词启发式
  const lowerInput = input.toLowerCase()
  if (KEYWORDS_TASK.some(kw => lowerInput.includes(kw))) return 'task'
  if (KEYWORDS_TOOL.some(kw => lowerInput.includes(kw))) return 'tool_use'
  return 'chat'
}
```

```typescript
import crypto from 'crypto'
import type { Message, StreamEvent, ToolCallData } from '../types.js'
import type { OrchestratorEngine, OrchestratorContext } from './engine.js'
import { OrchestratorState } from './states.js'

export class StateMachineOrchestrator implements OrchestratorEngine {
  private maxIterations: number

  constructor(maxIterations: number = 100) {
    this.maxIterations = maxIterations
  }

  async run(userInput: string, context: OrchestratorContext): Promise<string> {
    const messages: Message[] = [...context.history, { role: 'user', content: userInput }]
    let iterations = 0

    while (iterations < this.maxIterations) {
      iterations++

      // Planning
      if (context.hooks) {
        await context.hooks.fire('orch_plan', { iteration: iterations })
      }

      // 调用 LLM
      const response = await context.llmClient!.chat(messages, context.model, context.toolRunner?.getToolSchemas())

      // 如果没有工具调用，直接返回
      if (!response.tool_calls || response.tool_calls.length === 0) {
        return response.content
      }

      // 执行工具
      messages.push({ role: 'assistant', content: response.content, tool_calls: response.tool_calls })

      for (const toolCall of response.tool_calls) {
        if (context.hooks) {
          await context.hooks.fire('tool_exec', { tool: toolCall.name, args: toolCall.arguments })
        }

        const result = await context.toolRunner!.run(toolCall.name, toolCall.arguments)
        messages.push({
          role: 'tool',
          content: result,
          tool_call_id: toolCall.id,
          tool_name: toolCall.name,
        })
      }
    }

    return 'Max iterations reached'
  }

  async *runStream(userInput: string, context: OrchestratorContext): AsyncGenerator<StreamEvent> {
    const messages: Message[] = [...context.history, { role: 'user', content: userInput }]
    let iterations = 0

    while (iterations < this.maxIterations) {
      iterations++

      // Planning
      if (context.hooks) {
        await context.hooks.fire('orch_plan', { iteration: iterations })
      }

      // 调用 LLM（流式）
      let hasToolCalls = false
      const toolCalls: ToolCallData[] = []
      let content = ''

      for await (const event of context.llmClient!.chatStream(messages, context.model, context.toolRunner?.getToolSchemas())) {
        if (event.type === 'text_delta') {
          content += event.text
          yield event
        } else if (event.type === 'tool_call') {
          hasToolCalls = true
          toolCalls.push({ id: event.id, name: event.name, arguments: event.input })
        }
      }

      // 如果没有工具调用，结束
      if (!hasToolCalls) {
        return
      }

      // 执行工具
      messages.push({ role: 'assistant', content, tool_calls: toolCalls })

      for (const toolCall of toolCalls) {
        if (context.hooks) {
          await context.hooks.fire('tool_exec', { tool: toolCall.name, args: toolCall.arguments })
        }

        const startTime = performance.now()
        try {
          const result = await context.toolRunner!.run(toolCall.name, toolCall.arguments)
          const elapsed = performance.now() - startTime
          messages.push({
            role: 'tool',
            content: result,
            tool_call_id: toolCall.id,
            tool_name: toolCall.name,
          })
          yield {
            type: 'tool_result',
            id: toolCall.id,
            name: toolCall.name,
            result,
            ms: elapsed,
            is_error: false,
          }
        } catch (e) {
          const elapsed = performance.now() - startTime
          const error = e instanceof Error ? e.message : String(e)
          messages.push({
            role: 'tool',
            content: `Error: ${error}`,
            tool_call_id: toolCall.id,
            tool_name: toolCall.name,
          })
          yield {
            type: 'tool_result',
            id: toolCall.id,
            name: toolCall.name,
            result: error,
            ms: elapsed,
            is_error: true,
          }
        }
      }
    }
  }
}
```

```typescript
import type { Tool } from './base.js'
import type { ToolRunner as ToolRunnerProtocol } from '../orchestrator/engine.js'

export class ToolRunner implements ToolRunnerProtocol {
  private tools: Map<string, Tool> = new Map()

  register(tool: Tool): void {
    this.tools.set(tool.name, tool)
  }

  async run(toolName: string, params: Record<string, unknown>): Promise<string> {
    const tool = this.tools.get(toolName)
    if (!tool) throw new Error(`Unknown tool: ${toolName}`)
    return await tool.execute(params)
  }

  getToolSchemas(): Record<string, unknown>[] {
    return [...this.tools.values()].map(t => t.toOpenAiSchema())
  }
}
```

```typescript
import { BaseTool } from '../base.js'

export class CalculatorTool extends BaseTool {
  readonly name = 'calculator'
  readonly description = 'Calculate mathematical expressions'
  readonly inputSchema = {
    type: 'object',
    properties: {
      expression: { type: 'string', description: 'Mathematical expression to evaluate' },
    },
    required: ['expression'],
  }

  async execute(params: Record<string, unknown>): Promise<string> {
    const expression = params.expression as string
    try {
      const result = this.safeEval(expression)
      return String(result)
    } catch (e) {
      return `Error: ${e instanceof Error ? e.message : String(e)}`
    }
  }

  private safeEval(expression: string): number {
    // 简单的递归下降解析器
    const tokens = this.tokenize(expression)
    const result = this.parseExpression(tokens, { pos: 0 })
    return result
  }

  private tokenize(expression: string): string[] {
    const tokens: string[] = []
    let i = 0
    while (i < expression.length) {
      if (/\s/.test(expression[i])) {
        i++
        continue
      }
      if (/[0-9.]/.test(expression[i])) {
        let num = ''
        while (i < expression.length && /[0-9.]/.test(expression[i])) {
          num += expression[i++]
        }
        tokens.push(num)
        continue
      }
      if (/[+\-*/%^()]/.test(expression[i])) {
        tokens.push(expression[i++])
        continue
      }
      throw new Error(`Invalid character: ${expression[i]}`)
    }
    return tokens
  }

  private parseExpression(tokens: string[], ctx: { pos: number }): number {
    let result = this.parseTerm(tokens, ctx)
    while (ctx.pos < tokens.length && (tokens[ctx.pos] === '+' || tokens[ctx.pos] === '-')) {
      const op = tokens[ctx.pos++]
      const right = this.parseTerm(tokens, ctx)
      if (op === '+') result += right
      else result -= right
    }
    return result
  }

  private parseTerm(tokens: string[], ctx: { pos: number }): number {
    let result = this.parseFactor(tokens, ctx)
    while (ctx.pos < tokens.length && (tokens[ctx.pos] === '*' || tokens[ctx.pos] === '/' || tokens[ctx.pos] === '%')) {
      const op = tokens[ctx.pos++]
      const right = this.parseFactor(tokens, ctx)
      if (op === '*') result *= right
      else if (op === '/') result /= right
      else result %= right
    }
    return result
  }

  private parseFactor(tokens: string[], ctx: { pos: number }): number {
    if (tokens[ctx.pos] === '-') {
      ctx.pos++
      return -this.parseFactor(tokens, ctx)
    }
    if (tokens[ctx.pos] === '+') {
      ctx.pos++
      return this.parseFactor(tokens, ctx)
    }
    return this.parsePower(tokens, ctx)
  }

  private parsePower(tokens: string[], ctx: { pos: number }): number {
    let result = this.parsePrimary(tokens, ctx)
    if (ctx.pos < tokens.length && tokens[ctx.pos] === '**') {
      ctx.pos++
      const right = this.parseFactor(tokens, ctx)
      if (right > 1000) throw new Error('Exponent too large')
      result = Math.pow(result, right)
    }
    return result
  }

  private parsePrimary(tokens: string[], ctx: { pos: number }): number {
    if (tokens[ctx.pos] === '(') {
      ctx.pos++
      const result = this.parseExpression(tokens, ctx)
      if (tokens[ctx.pos] !== ')') throw new Error('Missing closing parenthesis')
      ctx.pos++
      return result
    }
    const num = parseFloat(tokens[ctx.pos++])
    if (isNaN(num)) throw new Error('Invalid number')
    return num
  }
}
```

```typescript
import fs from 'fs'
import path from 'path'
import { BaseTool } from '../base.js'

export class FileReadTool extends BaseTool {
  readonly name = 'file_read'
  readonly description = 'Read file contents'
  readonly inputSchema = {
    type: 'object',
    properties: {
      path: { type: 'string', description: 'File path to read' },
    },
    required: ['path'],
  }

  private allowedRoot: string

  constructor(allowedRoot: string = process.cwd()) {
    super()
    this.allowedRoot = path.resolve(allowedRoot)
  }

  async execute(params: Record<string, unknown>): Promise<string> {
    const filePath = path.resolve(params.path as string)

    // 检查路径是否在 allowed_root 内
    if (!filePath.startsWith(this.allowedRoot)) {
      return 'Error: Access denied - path outside allowed directory'
    }

    // 检查符号链接
    try {
      const stat = fs.lstatSync(filePath)
      if (stat.isSymbolicLink()) {
        return 'Error: Access denied - symbolic links not allowed'
      }
    } catch {
      return `Error: File not found: ${params.path}`
    }

    // 读取文件（限制 100KB）
    try {
      const content = fs.readFileSync(filePath, 'utf-8')
      if (content.length > 100 * 1024) {
        return content.slice(0, 100 * 1024) + '\n... (truncated at 100KB)'
      }
      return content
    } catch (e) {
      return `Error reading file: ${e instanceof Error ? e.message : String(e)}`
    }
  }
}
```

```typescript
import path from 'path'
import crypto from 'crypto'
import { BaseTool } from '../base.js'

export class FileWriteTool extends BaseTool {
  readonly name = 'file_write'
  readonly description = 'Write content to a file'
  readonly inputSchema = {
    type: 'object',
    properties: {
      path: { type: 'string', description: 'File path to write' },
      content: { type: 'string', description: 'Content to write' },
    },
    required: ['path', 'content'],
  }

  private dataDir: string

  constructor(dataDir: string) {
    super()
    this.dataDir = path.resolve(dataDir)
  }

  async execute(params: Record<string, unknown>): Promise<string> {
    const filePath = path.resolve(params.path as string)
    const content = params.content as string

    // 检查路径是否在 data_dir 内
    if (!filePath.startsWith(this.dataDir)) {
      return 'Error: Access denied - can only write to data directory'
    }

    // 检查符号链接
    try {
      const stat = fs.lstatSync(filePath)
      if (stat.isSymbolicLink()) {
        return 'Error: Access denied - symbolic links not allowed'
      }
    } catch {
      // 文件不存在，继续
    }

    // 确保目录存在
    const dir = path.dirname(filePath)
    fs.mkdirSync(dir, { recursive: true })

    // 原子写入
    const tmpPath = path.join(dir, `.tmp-${crypto.randomUUID()}`)
    const fd = fs.openSync(tmpPath, 'w')
    fs.writeSync(fd, content)
    fs.fsyncSync(fd)
    fs.closeSync(fd)
    fs.renameSync(tmpPath, filePath)

    return `File written: ${params.path}`
  }
}
```

```typescript
import fs from 'fs'
import path from 'path'
import crypto from 'crypto'
import { BaseTool } from '../base.js'

export class FileEditTool extends BaseTool {
  readonly name = 'file_edit'
  readonly description = 'Edit file by replacing text'
  readonly inputSchema = {
    type: 'object',
    properties: {
      path: { type: 'string', description: 'File path to edit' },
      old_text: { type: 'string', description: 'Text to replace' },
      new_text: { type: 'string', description: 'New text' },
    },
    required: ['path', 'old_text', 'new_text'],
  }

  private dataDir: string

  constructor(dataDir: string) {
    super()
    this.dataDir = path.resolve(dataDir)
  }

  async execute(params: Record<string, unknown>): Promise<string> {
    const filePath = path.resolve(params.path as string)
    const oldText = params.old_text as string
    const newText = params.new_text as string

    // 检查路径是否在 data_dir 内
    if (!filePath.startsWith(this.dataDir)) {
      return 'Error: Access denied - can only edit files in data directory'
    }

    // 读取文件
    let content: string
    try {
      content = fs.readFileSync(filePath, 'utf-8')
    } catch {
      return `Error: File not found: ${params.path}`
    }

    // 检查唯一匹配
    const count = content.split(oldText).length - 1
    if (count === 0) {
      return 'Error: old_text not found in file'
    }
    if (count > 1) {
      return `Error: old_text found ${count} times, must be unique`
    }

    // 替换
    const newContent = content.replace(oldText, newText)

    // 原子写入
    const dir = path.dirname(filePath)
    const tmpPath = path.join(dir, `.tmp-${crypto.randomUUID()}`)
    const fd = fs.openSync(tmpPath, 'w')
    fs.writeSync(fd, newContent)
    fs.fsyncSync(fd)
    fs.closeSync(fd)
    fs.renameSync(tmpPath, filePath)

    return `File edited: ${params.path}`
  }
}
```
