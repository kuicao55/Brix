# LangGraph Orchestrator Implementation Plan

> **Harness note:** This plan is executed via `harness-execution` using the Orchestrator / Executor / Reviewer architecture. Each task goes through Executor (TDD implementation) → Spec Reviewer (compliance check) → Code Quality Reviewer (adversarial review). Only Code Quality Review PASS closes a task.

**Goal:** Add a LangGraph-based orchestrator that implements the same `OrchestratorEngine` protocol, allowing users to switch between the pure Python state machine and LangGraph execution.

**Milestone ref:** milestone-2 from claude-progress.json

**Architecture:** The `LangGraphOrchestrator` uses LangGraph's `StateGraph` to define the same state transitions as the pure Python version (PLANNING → EXECUTING → REVIEWING → RESPONDING). It implements the `OrchestratorEngine` protocol so it's a drop-in replacement. A config option or constructor parameter controls which engine to use.

**Tech Stack:** langgraph, langchain-core

```python
# orchestrator/langgraph_engine.py
from __future__ import annotations

from typing import Any, TypedDict

from langgraph.graph import END, StateGraph

from orchestrator.engine import OrchestratorContext


class AgentState(TypedDict):
    messages: list[dict]
    tool_calls: list[Any]
    response: str
    iteration: int
    done: bool


class LangGraphOrchestrator:
    """LangGraph-based orchestrator implementing OrchestratorEngine protocol."""

    def __init__(self, max_iterations: int = 5):
        self.max_iterations = max_iterations
        self._graph = self._build_graph()

    def _build_graph(self) -> StateGraph:
        graph = StateGraph(AgentState)

        graph.add_node("plan", self._plan_node)
        graph.add_node("execute", self._execute_node)
        graph.add_node("review", self._review_node)
        graph.add_node("respond", self._respond_node)

        graph.set_entry_point("plan")
        graph.add_conditional_edges(
            "plan",
            self._should_execute_or_respond,
            {"execute": "execute", "respond": "respond"},
        )
        graph.add_edge("execute", "review")
        graph.add_conditional_edges(
            "review",
            self._should_replan_or_respond,
            {"plan": "plan", "respond": "respond"},
        )
        graph.add_edge("respond", END)

        return graph.compile()

    async def run(self, user_input: str, context: OrchestratorContext) -> str:
        initial_state: AgentState = {
            "messages": [{"role": "user", "content": user_input}],
            "tool_calls": [],
            "response": "",
            "iteration": 0,
            "done": False,
        }

        # Store context for node access
        self._context = context

        result = await self._graph.ainvoke(initial_state)
        return result.get("response", "")

    async def _plan_node(self, state: AgentState) -> dict:
        tool_schemas = self._context.tool_runner.get_tool_schemas()
        llm_response = await self._context.llm_client.chat(
            messages=state["messages"],
            model=self._context.model,
            tools=tool_schemas if tool_schemas else None,
        )

        new_messages = list(state["messages"])
        if llm_response.tool_calls:
            new_messages.append({
                "role": "assistant",
                "content": llm_response.content,
                "tool_calls": llm_response.tool_calls,
            })
            return {
                "messages": new_messages,
                "tool_calls": llm_response.tool_calls,
            }
        else:
            new_messages.append({
                "role": "assistant",
                "content": llm_response.content,
            })
            return {
                "messages": new_messages,
                "tool_calls": [],
                "response": llm_response.content,
            }

    async def _execute_node(self, state: AgentState) -> dict:
        new_messages = list(state["messages"])
        for tc in state["tool_calls"]:
            result = await self._context.tool_runner.run(tc.name, tc.arguments)
            new_messages.append({
                "role": "tool",
                "tool_name": tc.name,
                "content": result,
            })
        return {"messages": new_messages, "tool_calls": []}

    async def _review_node(self, state: AgentState) -> dict:
        return {"iteration": state["iteration"] + 1}

    def _should_execute_or_respond(self, state: AgentState) -> str:
        if state["tool_calls"]:
            return "execute"
        return "respond"

    def _should_replan_or_respond(self, state: AgentState) -> str:
        if state["iteration"] >= self.max_iterations:
            return "respond"
        if state["tool_calls"]:
            return "plan"
        return "respond"

    async def _respond_node(self, state: AgentState) -> dict:
        # Extract response from last assistant message if not already set
        if not state["response"]:
            for msg in reversed(state["messages"]):
                if msg.get("role") == "assistant" and msg.get("content"):
                    return {"response": msg["content"]}
        return {}
```
