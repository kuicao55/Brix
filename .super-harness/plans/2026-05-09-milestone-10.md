```python
# cli/tool_display.py

from cli.spinner import Spinner

class ToolDisplay:
    def __init__(self, console: Console) -> None:
        self._console = console
        self._active_spinner: Spinner | None = None  # 新增

    def show_tool_start(self, tool_name: str, tool_input: dict) -> None:
        # 原有逻辑：打印静态面板
        self._console.print(Text("  ⏺ Calling tools...", style="dim"))
        panel = Panel(...)
        self._console.print(panel)
        # 新增：启动 Spinner
        self._active_spinner = Spinner(self._console, label="Calling tools...")
        self._active_spinner.start()

    def show_tool_result(self, tool_name: str, result: str, elapsed_ms: float,
                         is_error: bool = False) -> None:
        # 新增：停止 Spinner
        if self._active_spinner is not None:
            self._active_spinner.stop()
            self._active_spinner = None
        # 原有逻辑：打印结果行
        ...

    def cleanup(self) -> None:
        """安全清理：异常路径调用，确保 spinner 不泄漏"""
        if self._active_spinner is not None:
            self._active_spinner.stop()
            self._active_spinner = None
```
